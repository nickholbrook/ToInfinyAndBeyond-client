#!/bin/bash
apt-get update -y

update-rc.d apache2 enable

git clone https://github.com/nickholbrook/ToInfinyAndBeyond-server.git /var/www/flask
cd /var/www/flask/
export FLASK_APP=hello.py
/usr/local/bin/flask run --host 0.0.0.0
